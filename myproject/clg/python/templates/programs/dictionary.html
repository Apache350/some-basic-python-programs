1.- Some basic programs of dictionary

#Syntax:- dict_name = {key1:value1, key2:value2,â€¦}

##creatin a dictionary:
stu = {101: 'Rahul', 102: 'Raj', 103: 'Sonam'}
fees = {'rahul':2000, 'raj':3000, 'sonam':8000 }
print(stu)
print(fees['raj'])

#crating a empty dictionary
fees={}
print(fees)

#modifying a dictionary--
stu = {101: 'Rahul', 102: 'Raj', 103: 'Sonam' }
stu[102] = 'Python'
print(stu)

#inserting/adding a entry--
stu = {101: 'Rahul', 102: 'Raj', 103: 'Sonam' }
stu[104] = 'Hello World'
print(stu)

#update a entry--
stu = {101: 'Rahul', 102: 'Raj', 103: 'Sonam' }
stu.update({101: 'Gupchup',105:'vishal'})
print(stu)

#Deleting an item
stu = {101: 'Rahul', 102: 'Raj', 103: 'Sonam' }
del stu[102]
print(stu)

#testing a key is present or not--
stu = {101: 'Rahul', 102: 'Raj', 103: 'Sonam' }
print(101 in stu) # If exists returns True
print(104 not in stu) # If not exists returns True

#clear()--use to remove all elements of a dic
stu = {101: 'Rahul', 102: 'Raj', 103: 'Sonam' }
stu.clear()
print(stu)
