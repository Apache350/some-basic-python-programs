1.-Some basic programs of list

#creating a list
a=[1,2,'vishal','#']
print(a)

# creating a empty list
a=[]
print(a)

###insert a value in list --
#append()--use to insert a value in end of a list
a=[1,2,3,"aa"]
a.append(10)
print(a)

#insert()--use to insert a value at a particular index
a=[1,2,9,4,"aa"]
a.insert(3,20)
print(a)

###deleting a value in list---
#pop()--use to delete last value of list
a=[1,2,9,4,"aa"]
a.pop()
print(a)

#pop(n)--use to delete a particular value index
a=[1,2,9,4,"aa"]
a.pop(3)
print(a)

#de(n)--use to delete a particular index value
a=[1,2,9,4,"aa"]
del a[2]
print(a)

##Sum of list

print(sum(a))
